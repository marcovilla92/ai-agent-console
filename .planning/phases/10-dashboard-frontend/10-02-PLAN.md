   ```javascript
   function taskDetail(taskId) {
     return {
       task: {},
       logText: '',
       approvalPending: false,
       approvalContext: '',
       agentOutputs: [],
       ws: null,

       async init() {
         await this.loadTask();
         await this.loadOutputs();
         this.connectWS();
       },

       async loadTask() {
         const resp = await fetch(`/tasks/${taskId}`, {credentials: 'same-origin'});
         if (resp.ok) this.task = await resp.json();
       },

       async loadOutputs() {
         // Agent outputs are included if the API supports it;
         // for now, load task and display what we have
         // Future: dedicated endpoint for agent_outputs
       },

       connectWS() {
         // Build WS token from browser's cached Basic Auth credentials
         // Since browser caches Basic Auth, we need to extract or prompt
         // Approach: prompt once, store in sessionStorage
         let token = sessionStorage.getItem('ws_token');
         if (!token) {
           const user = prompt('Username:');
           const pass = prompt('Password:');
           token = btoa(user + ':' + pass);
           sessionStorage.setItem('ws_token', token);
         }
         const proto = location.protocol === 'https:' ? 'wss:' : 'ws:';
         const url = `${proto}//${location.host}/ws/tasks/${taskId}?token=${token}`;
         this.ws = new WebSocket(url);

         this.ws.onmessage = (e) => {
           const msg = JSON.parse(e.data);
           if (msg.type === 'chunk') {
             this.logText += msg.data;
             this.$nextTick(() => {
               this.$refs.logContainer.scrollTop = this.$refs.logContainer.scrollHeight;
             });
           } else if (msg.type === 'status') {
             this.task.status = msg.data;
             if (['completed', 'failed', 'cancelled'].includes(msg.data)) {
               this.loadTask();  // Refresh full task data
             }
           } else if (msg.type === 'approval_required') {
             this.approvalPending = true;
             this.approvalContext = typeof msg.data === 'string' ? msg.data : JSON.stringify(msg.data, null, 2);
           }
           // Ignore "ping" type
         };

         this.ws.onclose = () => {
           // Reconnect after 3 seconds if task is still active
           setTimeout(() => {
             if (this.task.status === 'running' || this.task.status === 'queued' || this.task.status === 'awaiting_approval') {
               this.connectWS();
             }
           }, 3000);
         };
       },

       async sendApproval(decision) {
         await fetch(`/tasks/${taskId}/approve`, {
           method: 'POST',
           headers: {'Content-Type': 'application/json'},
           credentials: 'same-origin',
           body: JSON.stringify({decision})
         });
         this.approvalPending = false;
       },

       async cancelTask() {
         await fetch(`/tasks/${taskId}/cancel`, {
           method: 'POST',
           credentials: 'same-origin'
         });
         this.loadTask();
       }
     }
   }
   ```

Use `{% raw %}` around the entire `<script>` block to prevent Jinja2 from interpreting JS template literals with `${}` as Jinja2 expressions. For the `{{ task_id }}` in `x-data`, place it OUTSIDE the raw block since it IS a Jinja2 variable.

Handle the Jinja2/Alpine delimiter collision carefully: the `x-data="taskDetail({{ task_id }})"` attribute uses Jinja2 interpolation for task_id, but all Alpine.js `x-text`, `x-bind`, `x-for` directives use Alpine.js expressions (no `{{ }}` needed).
