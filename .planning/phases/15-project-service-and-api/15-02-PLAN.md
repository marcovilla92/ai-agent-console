       ```python
       async def create_project(self, name: str, description: str = "", template: str = "blank") -> Project:
           slug = sanitize_project_name(name)
           path = str(self.WORKSPACE_ROOT / slug)
           # 1. Create base folder (reuse create_project from project.py -- BUT we override since template handling is different)
           project_dir = Path(path)
           if project_dir.exists():
               raise FileExistsError(f"Project folder already exists: {project_dir}")
           project_dir.mkdir(parents=True, exist_ok=True)
           # 2. Scaffold from template
           scaffold_from_template(template, project_dir, {"name": name, "slug": slug, "description": description, "date": datetime.now().strftime("%Y-%m-%d"), "author": "ubuntu"})
           # 3. git init + initial commit (async subprocess with 10s timeout)
           await git_init_project(str(project_dir))
           # 4. Insert DB record
           now = datetime.now(timezone.utc)
           project = Project(name=name, slug=slug, path=path, description=description, created_at=now)
           project.id = await self._repo.insert(project)
           # 5. Emit event
           await emit_event(ProjectEvent.PROJECT_CREATED, {"id": project.id, "name": name, "path": path})
           return project
       ```
