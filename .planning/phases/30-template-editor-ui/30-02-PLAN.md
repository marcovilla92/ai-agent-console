Add method `toggleTemplateEditMode()`:
```javascript
toggleTemplateEditMode() {
  this.templateEditMode = !this.templateEditMode;
  if (this.templateEditMode) {
    // Enter edit mode -- copy current files as baseline
    this.templateEditedFiles = {};
    this.templateNewFiles = {};
    this.templateDeletedFiles = [];
  } else {
    // Exit edit mode -- discard changes
    this.templateEditedFiles = {};
    this.templateNewFiles = {};
    this.templateDeletedFiles = [];
  }
}
```

Add method `updateFileContent(path, content)`:
```javascript
updateFileContent(path, content) {
  // Only track if different from original
  if (this.templateFiles[path] !== undefined && content === this.templateFiles[path]) {
    delete this.templateEditedFiles[path];
  } else {
    this.templateEditedFiles[path] = content;
  }
}
```

**Add `saveTemplateFiles()` method:**
```javascript
async saveTemplateFiles() {
  var id = this.selectedTemplate.id;
  var payload = {};

  // Merge edited and new files into files_upsert
  var upsert = {};
  for (var path in this.templateEditedFiles) {
    upsert[path] = this.templateEditedFiles[path];
  }
  for (var path in this.templateNewFiles) {
    upsert[path] = this.templateNewFiles[path];
  }
  if (Object.keys(upsert).length > 0) {
    payload.files_upsert = upsert;
  }
  if (this.templateDeletedFiles.length > 0) {
    payload.files_delete = this.templateDeletedFiles;
  }

  this.submitting = true;
  try {
    var resp = await apiFetch('/templates/' + id, {
      method: 'PUT',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(payload)
    });
    if (resp.ok) {
      // Reload files to reflect saved state
      await this.loadTemplateFiles(id);
      this.templateEditMode = false;
      this.templateEditedFiles = {};
      this.templateNewFiles = {};
      this.templateDeletedFiles = [];
      this.templateShowPreview = false;
      this.error = null;
    } else {
      var err = await resp.json().catch(function() { return {}; });
      this.error = err.detail || 'Failed to save template files';
    }
  } catch(e) {
    this.error = 'Failed to save template files';
  }
  this.submitting = false;
}
```
